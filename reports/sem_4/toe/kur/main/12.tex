\section{Определение в «замкнутой» форме вынужденной составляющей реакции при периодическом входном сигнале}
\label{sec:zamkn}

Расчёт реакции в замкнутой форме позволяет получить точное аналитическое выражение для установившегося режима без использования бесконечных рядов. Для получения аналитического выражения применим операторный метод. 

Введем функцию $\mu(t)$, описывающую форму заданного периодического сигнала. Тогда входной ток $i_{1}(t) = I_m \mu(t)$. Первый период $\mu(t)$ описывается как:
\begin{equation}
    \mu_{01}(t) = \delta_1(t) - 2\delta_1\left(t - \frac{T}{4}\right) + \delta_1\left(t - \frac{T}{2}\right).
\end{equation}

Изображение этого одиночного импульса по Лапласу:
\begin{equation}
    \Mu_{01}(s) = \frac{1}{s} \left( 1 - 2e^{-sT/4} + e^{-sT/2} \right) = \frac{1}{s} \left( 1 - e^{-sT/4} \right)^2.
\end{equation}

Согласно свойствам преобразования Лапласа, изображение периодического сигнала формируется умножением изображения его первого периода на $\frac{1}{1 - e^{-sT}}$. Изображение входного тока:
\begin{equation}
    I_{1}(s) = I_m \frac{\Mu_{01}(s)}{1 - e^{-sT}} = \frac{I_m}{s} \frac{\left( 1 - e^{-sT/4} \right)^2}{1 - e^{-sT}}.
\end{equation}

Изображение реакции цепи определяется умножением на передаточную функцию $H(s)$:
\begin{equation}
    I_{2}(s) = H(s) I_{1}(s) = I_m \frac{H(s)}{s} \frac{\left( 1 - e^{-sT/4} \right)^2}{1 - e^{-sT}}.
\end{equation}

Разложим дробь $\frac{H(s)}{s}$ на простые составляющие, используя теорему разложения:
\begin{equation}
    \frac{H(s)}{s} = \frac{H(0)}{s} + \sum_{k=1}^N \frac{A_k / p_k}{s - p_k},
\end{equation}
где $p_k$ — полюсы передаточной функции, $A_k$ — вычеты $H(s)$ в этих полюсах, $H(0)$ — вычет в нуле.

Подставим разложение в выражение для реакции и сгруппируем слагаемые:
\begin{equation}
    I_{2}(s) = I_m \frac{H(0)}{s} \frac{\left( 1 - e^{-sT/4} \right)^2}{1 - e^{-sT}} + I_m \sum_{k=1}^N \frac{A_k}{p_k} \left[ \frac{1}{s - p_k} \frac{\left( 1 - e^{-sT/4} \right)^2}{1 - e^{-sT}} \right];
\end{equation}
\begin{equation}
    I_{2}(s) = I_m H(0) \left[ \frac{1}{s} \frac{\left( 1 - e^{-sT/4} \right)^2}{1 - e^{-sT}} \right] + I_m \sum_{k=1}^N \frac{A_k}{p_k} \left[ \frac{1}{s - p_k} \frac{\left( 1 - e^{-sT/4} \right)^2}{1 - e^{-sT}} \right].
\end{equation}

Выполним переход в область оригинала для каждого слагаемого в отдельности в пределах первого периода. Выражение в первых квадратных скобках представляет собой изображение исходного сигнала $\mu(t)$. Следовательно, первое слагаемое дает реакцию $I_m H(0) \mu(t)$.

% Второе слагаемое. Дробь $\frac{1}{s - p_k} \left( 1 - e^{-sT/4} \right)^2$ является изображением реакции на одиночный импульс. По теореме запаздывания ее оригинал на интервале $t \in [0, T)$ представляет собой функцию $\eta_k(t)$:
% \begin{equation}
%     \eta_k(t) = e^{p_k t}\delta_1(t) - 2e^{p_k(t - T/4)}\delta_1(t - \frac{T}4) + e^{p_k(t - T/2)}\delta_1(t - \frac T2).
% \end{equation}

% Деление на $1 - e^{-sT}$ в операторной области эквивалентно добавлению во временной области бесконечной суммы затухающих откликов от импульсов предыдущих периодов. К моменту окончания одного периода ($t=T$) реакция достигает значения $\eta_k(T) = e^{p_k T} \left(1 - e^{-p_k T/4}\right)^2$. Сумма убывающей геометрической прогрессии этих остаточных значений для момента времени $t$ внутри текущего периода дает величину $\frac{\eta_k(T) e^{p_k t}}{1 - e^{p_k T}}$.

% В результате точная вынужденная реакция на интервале $t \in [0, T)$ определяется суперпозицией квазистатической части, отклика на текущий импульс $\eta_k(t)$ и суммы остаточных откликов:
% \begin{equation}
%     i_{2}(t) = I_m H(0) \mu(t) + I_m \sum_{k=1}^{N} \frac{A_k}{p_k} \left[ \frac{e^{p_k (t + T)}}{1 - e^{p_k T}} \left(1 - e^{-p_k T/4}\right)^2 + \eta_k(t) \right],
%     \label{form:zamkn}
% \end{equation}
Второе слагаемое описывает переходные процессы. Дробь $\frac{1}{s - p_k} \left( 1 - e^{-sT/4} \right)^2$ --- это изображение реакции цепи только на один одиночный импульс. Обозначим её оригинал как $\eta_k(t)$. По теореме запаздывания на интервале $t \in [0, T)$ она равна:
\begin{equation}
    \eta_k(t) = e^{p_k t}\delta_1(t) - 2e^{p_k(t - T/4)}\delta_1(t - T/4) + e^{p_k(t - T/2)}\delta_1(t - T/2).
\end{equation}

Но в цепи установившийся режим, то есть импульсы подавались и до момента $t=0$. Деление на $1 - e^{-sT}$ в операторной области означает, что во временной области мы должны сложить отклики от всех прошлых импульсов. 

К концу своего периода (при $t=T$) одиночный импульс оставляет после себя остаточное значение $\eta_k(T) = e^{p_k T} \left(1 - e^{-p_k T/4}\right)^2$. 

Поскольку свободные процессы затухают по закону $e^{p_k t}$, каждый предыдущий импульс добавляет к текущему моменту времени $t$ свой затухающий «хвост». Сумма всех этих бесконечных хвостов образует убывающую геометрическую прогрессию, сумма которой равна $\frac{\eta_k(T) e^{p_k t}}{1 - e^{p_k T}}$. 

Подставив сюда $\eta_k(T)$, получим:
\begin{equation}
   \frac{e^{p_k T} \left(1 - e^{-p_k T/4}\right)^2 e^{p_k t}}{1 - e^{p_k T}} = \frac{e^{p_k (t + T)}}{1 - e^{p_k T}} \left(1 - e^{-p_k T/4}\right)^2. 
\end{equation}

Складывая реакцию на текущий импульс $\eta_k(t)$ и сумму хвостов от прошлых импульсов, получаем итоговую формулу для тока на интервале $t \in [0, T)$:
\begin{equation}
    i_{2}(t) = I_m H(0) \mu(t) + I_m \sum_{k=1}^{N} \frac{A_k}{p_k} \left[ \frac{e^{p_k (t + T)}}{1 - e^{p_k T}} \left(1 - e^{-p_k T/4}\right)^2 + \eta_k(t) \right],
    \label{form:zamkn}
\end{equation}
где $\mu(t)$ — нормированная форма входного сигнала (принимает значения $1, -1, 0$), $p_k$ — полюсы передаточной функции, $A_k$ — соответствующие вычеты.

Функция $\eta_k(t)$ учитывает экспоненциальную составляющую от скачков внутри текущего периода:
$$ \eta_k(t) = \begin{cases}
    e^{p_k t}, & 0 \le t < T/4 \\
    e^{p_k t} - 2e^{p_k(t - T/4)}, & T/4 \le t < T/2 \\
    e^{p_k t} - 2e^{p_k(t - T/4)} + e^{p_k(t - T/2)}, & T/2 \le t < T
\end{cases} $$
