\subsection*{ \fbox{Задание 3.} Расчет температурного коэффициента емкости $\alpha_C$}
\begin{quote} 
    \textit{Рассчитать значения температурного коэффициента емкости $\alpha_C$ для исследованных образцов, пользуясь \cref{form:alphac}. Значение производной $dC/dt$ найти путем графического дифференцирования построенной зависимости $C(t)$ для каждого образца.}
\end{quote}

Температурный коэффициент емкости (ТКЕ) определяется по формуле:
\begin{equation}
    \alpha_C = \frac{1}{C} \frac{dC}{dt}
\end{equation}

Для линейных диэлектриков (образцы 1--4) производная $dC/dt$ принимается постоянной на всем температурном диапазоне и рассчитывается по крайним точкам ($t_1$ и $t_N$). Для нелинейного диэлектрика --- сегнетокерамики --- применяется метод разностей для каждого интервала $\Delta t$. Истинная емкость образца рассчитывается как $C_{net} = C_{meas} - C_0$.

\subsection*{ \fbox{Задание 4.} Расчет температурного коэффициента диэлектрической проницаемости $\alpha_\varepsilon$}
\begin{quote} 
    \textit{Пользуясь формулой (5), рассчитать значения температурного коэффициента диэлектрической проницаемости $\alpha_\varepsilon$ для исследованных материалов. Значения $\alpha_{l\text{д}}$, необходимые для вычисления, указаны на испытательном стенде.}
\end{quote}

Температурный коэффициент диэлектрической проницаемости (ТКДП) рассчитывается по формуле:
\begin{equation}
    \alpha_\varepsilon = \alpha_C - \alpha_{l\text{д}}
\end{equation}

Ниже представлены подробные расчеты $\alpha_C$ и $\alpha_\varepsilon$ для всех образцов ($C_0 = \pv[-12.1]{p['C0']}$ пФ).

\textit{1. Неорганическое стекло}

$t_1 = \pv[0]{p['t'][0]}\,^\circ\text{C}$, $C_{meas}(t_1) = \pv[-12.0]{p['C1'][0]}$ пФ; $t_N = \pv[0]{p['t'][-1]}\,^\circ\text{C}$, $C_{meas}(t_N) = \pv[-12.0]{p['C1'][-1]}$ пФ.

$C_{net}(t_1) = \pv[-12.0]{p['C1'][0]} - \pv[-12.1]{p['C0']} = \pv[-12.1]{r['C1_net'][0]}$ пФ; $C_{net}(t_N) = \pv[-12.0]{p['C1'][-1]} - \pv[-12.1]{p['C0']} = \pv[-12.1]{r['C1_net'][-1]}$ пФ.
\begin{equation*}
    \alpha_C = \frac{1}{C_{net}(t_1)} \frac{C_{net}(t_N) - C_{net}(t_1)}{t_N - t_1} = \frac{1}{\pv[-12.1]{r['C1_net'][0]}} \frac{\pv[-12.1]{r['C1_net'][-1]} - \pv[-12.1]{r['C1_net'][0]}}{\pv[0]{p['t'][-1]} - \pv[0]{p['t'][0]}} = \pv[.2e]{(r['C1_net'][-1]-r['C1_net'][0])/(p['t'][-1]-p['t'][0])/r['C1_net'][0]}\; \text{K}^{-1}
\end{equation*}
\begin{equation*}
    \alpha_\varepsilon = \alpha_C - \alpha_{l\text{д}} = \pv[.2e]{(r['C1_net'][-1]-r['C1_net'][0])/(p['t'][-1]-p['t'][0])/r['C1_net'][0]} - \pv[.2e]{r['ald'][0]} = \pv[.2e]{(r['C1_net'][-1]-r['C1_net'][0])/(p['t'][-1]-p['t'][0])/r['C1_net'][0] - r['ald'][0]}\; \text{K}^{-1}
\end{equation*}

\textit{2. Слюда}

$C_{meas}(\pv[0]{p['t'][0]}\,^\circ\text{C}) = \pv[-12.0]{p['C2'][0]}$ пФ; $C_{meas}(\pv[0]{p['t'][-1]}\,^\circ\text{C}) = \pv[-12.0]{p['C2'][-1]}$ пФ.

$C_{net}(\pv[0]{p['t'][0]}\,^\circ\text{C}) = \pv[-12.1]{r['C2_net'][0]}$ пФ; $C_{net}(\pv[0]{p['t'][-1]}\,^\circ\text{C}) = \pv[-12.1]{r['C2_net'][-1]}$ пФ.
\begin{equation*}
    \alpha_C = \frac{1}{\pv[-12.1]{r['C2_net'][0]}} \frac{\pv[-12.1]{r['C2_net'][-1]} - \pv[-12.1]{r['C2_net'][0]}}{\pv[0]{p['t'][-1]} - \pv[0]{p['t'][0]}} = \pv[.2e]{(r['C2_net'][-1]-r['C2_net'][0])/(p['t'][-1]-p['t'][0])/r['C2_net'][0]}\; \text{K}^{-1}
\end{equation*}
\begin{equation*}
    \alpha_\varepsilon = \pv[.2e]{(r['C2_net'][-1]-r['C2_net'][0])/(p['t'][-1]-p['t'][0])/r['C2_net'][0]} - \pv[.3e]{r['ald'][1]} = \pv[.3e]{(r['C2_net'][-1]-r['C2_net'][0])/(p['t'][-1]-p['t'][0])/r['C2_net'][0] - r['ald'][1]}\; \text{K}^{-1}
\end{equation*}

\textit{3. Тиконд}

$C_{meas}(\pv[0]{p['t'][0]}\,^\circ\text{C}) = \pv[-12.0]{p['C3'][0]}$ пФ; $C_{meas}(\pv[0]{p['t'][-1]}\,^\circ\text{C}) = \pv[-12.0]{p['C3'][-1]}$ пФ.

$C_{net}(\pv[0]{p['t'][0]}\,^\circ\text{C}) = \pv[-12.1]{r['C3_net'][0]}$ пФ; $C_{net}(\pv[0]{p['t'][-1]}\,^\circ\text{C}) = \pv[-12.1]{r['C3_net'][-1]}$ пФ.
\begin{equation*}
    \alpha_C = \frac{1}{\pv[-12.1]{r['C3_net'][0]}} \frac{\pv[-12.1]{r['C3_net'][-1]} - \pv[-12.1]{r['C3_net'][0]}}{\pv[0]{p['t'][-1]} - \pv[0]{p['t'][0]}} = \pv[.2e]{(r['C3_net'][-1]-r['C3_net'][0])/(p['t'][-1]-p['t'][0])/r['C3_net'][0]}\; \text{K}^{-1}
\end{equation*}
\begin{equation*}
    \alpha_\varepsilon = \pv[.2e]{(r['C3_net'][-1]-r['C3_net'][0])/(p['t'][-1]-p['t'][0])/r['C3_net'][0]} - \pv[.2e]{r['ald'][2]} = \pv[.2e]{(r['C3_net'][-1]-r['C3_net'][0])/(p['t'][-1]-p['t'][0])/r['C3_net'][0] - r['ald'][2]}\; \text{K}^{-1}
\end{equation*}

\textit{4. Полипропилен}

$C_{meas}(\pv[0]{p['t'][0]}\,^\circ\text{C}) = \pv[-12.0]{p['C4'][0]}$ пФ; $C_{meas}(\pv[0]{p['t'][-1]}\,^\circ\text{C}) = \pv[-12.0]{p['C4'][-1]}$ пФ.

$C_{net}(\pv[0]{p['t'][0]}\,^\circ\text{C}) = \pv[-12.1]{r['C4_net'][0]}$ пФ; $C_{net}(\pv[0]{p['t'][-1]}\,^\circ\text{C}) = \pv[-12.1]{r['C4_net'][-1]}$ пФ.
\begin{equation*}
    \alpha_C = \frac{1}{\pv[-12.1]{r['C4_net'][0]}} \frac{\pv[-12.1]{r['C4_net'][-1]} - \pv[-12.1]{r['C4_net'][0]}}{\pv[0]{p['t'][-1]} - \pv[0]{p['t'][0]}} = \pv[.2e]{(r['C4_net'][-1]-r['C4_net'][0])/(p['t'][-1]-p['t'][0])/r['C4_net'][0]}\; \text{K}^{-1}
\end{equation*}
\begin{equation*}
    \alpha_\varepsilon = \pv[.2e]{(r['C4_net'][-1]-r['C4_net'][0])/(p['t'][-1]-p['t'][0])/r['C4_net'][0]} - \pv[.2e]{r['ald'][3]} = \pv[.2e]{(r['C4_net'][-1]-r['C4_net'][0])/(p['t'][-1]-p['t'][0])/r['C4_net'][0] - r['ald'][3]}\; \text{K}^{-1}
\end{equation*}

\textit{5. Сегнетокерамика (пример для первого температурного интервала)}

$t_1 = \pv[0]{p['t'][0]}\,^\circ\text{C}$, $C_{meas}(t_1) = \pv[-12.0]{p['C5'][0]}$ пФ; $t_2 = \pv[0]{p['t'][1]}\,^\circ\text{C}$, $C_{meas}(t_2) = \pv[-12.0]{p['C5'][1]}$ пФ.

$C_{net}(t_1) = \pv[-12.1]{r['C5_net'][0]}$ пФ; $C_{net}(t_2) = \pv[-12.1]{r['C5_net'][1]}$ пФ.
\begin{equation*}
    \alpha_{C,1} = \frac{1}{C_{net}(t_1)} \frac{C_{net}(t_2) - C_{net}(t_1)}{t_2 - t_1} = \frac{1}{\pv[-12.1]{r['C5_net'][0]}} \frac{\pv[-12.1]{r['C5_net'][1]} - \pv[-12.1]{r['C5_net'][0]}}{\pv[0]{p['t'][1]} - \pv[0]{p['t'][0]}} = \pv[.2e]{(r['C5_net'][1]-r['C5_net'][0])/(p['t'][1]-p['t'][0])/r['C5_net'][0]}\; \text{K}^{-1}

\begin{equation*}

\section*{Выводы}

На основании анализа рассчитанных температурных зависимостей $\alpha_\varepsilon(t)$ и графиков $C(t)$ установлены преобладающие механизмы поляризации для исследованных материалов:

\begin{enumerate}
    \item \textit{Полипропилен (неполярный полимер)}.
    Для данного материала наблюдается отрицательный температурный коэффициент:
    \begin{equation}
        \alpha_\varepsilon < 0
    \end{equation}
    Преобладающим механизмом является \textit{электронная поляризация}.% Процесс устанавливается практически мгновенно и не зависит от теплового движения. Снижение диэлектрической проницаемости при нагреве обусловлено исключительно тепловым расширением материала (уменьшением плотности), что приводит к снижению концентрации поляризуемых молекул в единице объема.

    \item \textit{Тиконд (рутиловая керамика)}.
    Наблюдается стабильно отрицательный коэффициент:
    \begin{equation}
        \alpha_\varepsilon < 0
    \end{equation}
    Преобладающий механизм --- \textit{электронная поляризация}.
    % ионной поляризацией. Отрицательный температурный коэффициент объясняется тем, что при нагревании кристаллическая решетка расширяется, внутреннее электрическое поле ослабевает, и ионная поляризуемость падает быстрее, чем растет объем.
    % При повышении температуры межионные расстояния в кристаллической решетке увеличиваются. Это приводит к ослаблению сил внутреннего электрического поля и снижению объемной поляризуемости. Эффект теплового расширения доминирует, вызывая спад емкости.

    \item \textit{Неорганическое стекло и слюда (неорганические диэлектрики с неплотной упаковкой ионов)}.
    Для этих материалов коэффициент положителен:
    \begin{equation}
        \alpha_\varepsilon > 0
    \end{equation}
    Помимо электронной и ионной, здесь выражена \textit{ионно-релаксационная поляризация}.% При повышении температуры тепловая энергия слабосвязанных ионов возрастает, что облегчает их переброс (релаксацию) между вакантными узлами под действием внешнего поля. Рост поляризуемости за счет увеличения подвижности ионов компенсирует эффект снижения плотности от теплового расширения, вследствие чего емкость возрастает.

    \item \textit{Сегнетокерамика (нелинейный диэлектрик)}.
    Наблюдается резкая нелинейность зависимости $C(t)$ с ярко выраженным максимумом при температуре $t \approx 44.0\,^\circ\text{C}$. Коэффициент $\alpha_\varepsilon$ меняет знак с положительного на отрицательный в точке экстремума.
    Преобладающий механизм --- \textit{спонтанная (доменная) поляризация}.
    При температурах ниже точки экстремума материал находится в сегнетоэлектрическом состоянии.% Рост температуры увеличивает подвижность доменных стенок, что ведет к росту $\varepsilon$.
    %Точка максимума является \textit{точкой Кюри} ($T_C$). При превышении этой температуры интенсивное тепловое движение разрушает спонтанную ориентацию диполей. Материал переходит в параэлектрическое состояние, и его диэлектрическая проницаемость падает в соответствии с законом Кюри-Вейса:
    %\begin{equation}
    %    \varepsilon = \frac{C_{CW}}{T - T_C}
    %\end{equation}
    %где $C_{CW}$ --- постоянная Кюри-Вейса.
\end{enumerate}
