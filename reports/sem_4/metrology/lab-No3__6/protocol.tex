\begin{luacode*}
function print_step_response_rows()
    local data = protocol.step_response
    local num_rows_data = #data.t_i
    local total_rows = 10
    
    for i = 1, num_rows_data do
        tex.print(string.format(
            "$%d$ & $%.2f$ & $%.2f$ & $%.2f$ \\\\",
            i,
            data.t_i[i] * 1e3, -- в мс
            data.u_in_i[i],
            data.u_out_i[i]
        ))
    end
    
    -- Добавляем пустые строки, если данных меньше 10
    for i = num_rows_data + 1, total_rows do
        tex.print(string.format("$%d$ & & & \\\\", i))
    end
end
\end{luacode*}
