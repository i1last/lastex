\begin{luacode*}
function csv2tblr(filename, num)
    local use_num = num or false

    local file = io.open(filename, 'r')
    local i = 0
    for line in file:lines() do
        i = i + 1
        local numeration = (use_num and tostring(i) .. " & ") or ""
        local data = line:gsub(';', ' & ') .. ' \\\\'
        tex.print(numeration .. data)
    end
    file:close()
end
\end{luacode*}
